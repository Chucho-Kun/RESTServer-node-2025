PETICIONEJEMPLO
GET

curl --location --request GET 'api/usuarios?q=hola&nombre=chucho-kun&api=234234234234%0A%0A' \ --header 'Content-Type: application/json' \ --data ''

POST

curl --location --request PUT 'api/usuarios' \ --header 'Content-Type: application/json' \ --data '{ "nombre" : "jesus", "edad" : "45" }'

curl --location --request PUT 'api/usuarios/10' \ --data ''

DELETE

curl --location --request DELETE 'api/usuarios' \ --data ''
